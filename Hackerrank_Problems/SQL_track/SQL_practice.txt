1. Weather Observation Station 1
SELECT CITY, STATE FROM STATION;

2. Weather Observation Station 3
SELECT DISTINCT CITY FROM STATION WHERE MOD(ID,2)=0 ;

3. Weather Observation Station 4
SELECT COUNT(CITY) - COUNT(DISTINCT CITY) FROM STATION;

4. Weather Observation Station 5
SELECT CITY, LENGTH(CITY) FROM (SELECT * FROM STATION 
WHERE LENGTH(CITY)=(SELECT MIN(LENGTH(CITY)) FROM STATION) ORDER BY CITY) WHERE ROWNUM=1;

SELECT CITY, LENGTH(CITY) FROM (SELECT * FROM STATION 
WHERE LENGTH(CITY)=(SELECT MAX(LENGTH(CITY)) FROM STATION) ORDER BY CITY) WHERE ROWNUM=1;

5. Weather Observation Station 6 
SELECT DISTINCT CITY FROM STATION WHERE LEFT(CITY,1) IN ('a', 'e', 'i', 'o', 'u','A','E','I','O','U');

6. Weather Observation Station 7
SELECT DISTINCT CITY FROM STATION WHERE RIGHT(CITY,1) IN ('a','e','i','o','u','A','E','I','O','U');

7. Weather Observation Station 8
SELECT DISTINCT CITY FROM STATION WHERE LEFT(CITY,1) IN ('a','e','i','o','u','A','E','I','O','U') AND RIGHT(CITY,1) in ('a','e','i','o','u','A','E','I','O','U');

8. Weather Observation Station 9 
SELECT DISTINCT CITY FROM STATION WHERE LEFT(CITY,1) NOT IN ('a','e','i','o','u','A','E','I','O','U');

(LEFT, RIGHT are for DB2)

9. Higher Than 75 Marks
SELECT NAME FROM STUDENTS WHERE MARKS > 75 ORDER BY RIGHT(NAME,3), ID;

10. Type of Triangle
SELECT CASE 
WHEN A + B <= C OR A + C <= B OR B + C <= A THEN 'Not A Triangle' 
WHEN A = B AND B = C THEN 'Equilateral' 
WHEN A = B OR B = C OR A = C THEN 'Isosceles' 
ELSE 'Scalene' 
END 
FROM TRIANGLES;

11. The PADS
SELECT NAME || '(' || SUBSTR(OCCUPATION,1,1) || ')' FROM OCCUPATIONS ORDER BY NAME;
SELECT 'There are a total of ' || COUNT(OCCUPATION) || ' ' || LOWER(OCCUPATION) || 's.' FROM OCCUPATIONS GROUP BY OCCUPATION ORDER BY COUNT(OCCUPATION), OCCUPATION;

12. Binary Tree Nodes
SELECT N, CASE WHEN P IS NULL THEN 'Root' 
WHEN N IN (SELECT P FROM BST) THEN 'Inner'
ELSE 'Leaf' END
FROM BST ORDER BY N;
